批量制作画册模板

    
X

在工作中经常会有若干个设计师共同完成一个任务。为了保证任务完成的一致性，需要制定一个统一标准的设计模板。本节将显示如何用脚本制定一个画册模板。

首先创建一个空白的脚本文档，并保存在硬盘上某个位置。

接着输入脚本代码：

//调用[Folder]的[selectDialog]命令，弹出文件夹选择窗口，提示用户选择输出模板文件的文件夹，并将文件夹存储在变量[outputFolder]中。
var outputFolder = Folder.selectDialog("选择输出的文件夹");

//定义一个变量[logo]，用来表示硬盘上的一张图片。它将被加载并放置在模板图像中的某个位置上。
var logo = File("E:\Logo.jpg");

//打开该图片，并将打开后的图像赋予变量[logoDoc]。
var logoDoc = open(logo);

//调用[selection]对象的[selectAll]命令，选择整张图片。
logoDoc.selection.selectAll();

//调用[selection]对象的[copy]命令，拷贝当前选区的内容至内存。
logoDoc.selection.copy();

//图片被拷贝至内存后，就不再需要这张图片了，可以把它关闭掉。
logoDoc.close(SaveOptions.DONOTSAVECHANGES);

//添加一个10次的循环，用来创建十页画册模板。您可以根据实际情况，设置适当的数字。
var pageCount = 10;
for(var i=1; i<=pageCount; i++)
{
	//设置模板的宽度和高度，在此设置为B5规格。
	var docWidth  = 2480;
	var docHeight = 3366;

	//设置模板的色彩模式和默认填充色。
	var mode = NewDocumentMode.RGB;
	var fill = DocumentFill.WHITE;

	//调用[documents]对象的[add]方法，创建一个空白模板文件，并将模板文件赋予[doc]变量。
	var doc = app.documents.add(docWidth, docHeight, 300, "page"+i, mode, fill);
    
	//在空白模板文件中，创建一个空白图层。
	var body = doc.artLayers.add();

	//调用[selection]对象的[select]方法，创建一个长方形选区并填充颜色。
	doc.selection.select([[0, 540], [docWidth, 540],[docWidth, 822], [0, 822]], SelectionType.REPLACE, 0, true);
	doc.selection.fill(getSolidColor(231, 121, 24));

	//使用同样的方式，创建另一个长方形区域并填充颜色。
	doc.selection.select([[0, 844], [docWidth, 844],[docWidth, 856], [0, 856]], SelectionType.REPLACE, 0, true);
	doc.selection.fill(getSolidColor(231, 121, 24));

	//使用同样的方式，创建另一个长方形区域并填充颜色。
	doc.selection.select([[0, 972], [docWidth, 972],[docWidth, 3018], [0, 3018]], SelectionType.REPLACE, 0, true);
	doc.selection.fill(getSolidColor(223, 223, 223));

	//判断当前页码是否属于奇数页。
	if(i % 2 != 0)
	{
		//奇数页的模板文件，创建多创建一个长方形。
		doc.selection.select([[0, 540], [720, 540],[720, 822], [0, 822]], SelectionType.REPLACE, 0, true);
		doc.selection.fill(getSolidColor(243, 182, 8));
	}

	//创建一个空白图层，并将图层属性设为文本图层，我们将页码放置在此图层上。
	var pageNumber = doc.artLayers.add();
	pageNumber.kind = LayerKind.TEXT;
	var textItemRef = pageNumber.textItem;
	textItemRef.contents = i;

	//将文本内容设置为遍历的数字（即页码）。并设置字体大小为18。
	textItemRef.size = 18;

	//判断如果是偶数页，页码在模板页面的右下角。如果是奇数页，页码放置在模板页面的左下角。
	if(i % 2 == 0)
		pageNumber.translate(docWidth - 420, docHeight - 460);
	else
		pageNumber.translate(-100, docHeight - 460);
        
	//将之前复制到内存的图标文件，复制到当前模板页面内，它将被复制到页面中心位置上。
	doc.paste();

	//判断如果是奇数页则把图标移至左上角，如果是偶数页，则把图标移至右上角。
	if(i % 2 != 0)
		doc.activeLayer.translate(-940, -1398);
	else
		doc.activeLayer.translate(964, -1398);
    
	//定义一个变量[file]，作为模板输出的文件。
	var file = new File(outputFolder + "/page" + i + ".jpg");

	//定义一个变量[jpegOptions]，用作图片输出的格式。
	var jpegOptions = new JPEGSaveOptions();

	//调用[doc]对象的[saveAs]方法，保存当前模板文件为JPEG图片。
	doc.saveAs(file, jpegOptions, true, Extension.LOWERCASE);
	doc.close(SaveOptions.DONOTSAVECHANGES);
}

//创建一个函数[getSolidColor]，用来生成相应的[SolidColor]对象。
function getSolidColor(r, g, b)
{
	//新建一个SolidColor对象，并赋予变量[sc]。
	var sc = new SolidColor();

	//然后设置[sc]对象的[rgb]属性的值。
	sc.rgb.red = r;
	sc.rgb.green = g;
	sc.rgb.blue = b;

	//在函数的结尾，将结果返回。
	return sc;
}

我们的脚本已经编写完成，把它保存下来。

然后切换到Photoshop。

接着依次点击[文件] > [脚本] > [浏览]，打开刚刚保存的脚本文件。

在弹出的[载入]窗口中，直接双击脚本名称，Photoshop将直接调用并执行该脚本。

点击选择模版导出后的文件夹。然后点击[确定]按钮，确认文件夹的选择。Photoshop开始批量导出画册模版。
